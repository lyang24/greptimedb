50/// Adapter trait for [`GrpcQueryHandler`] that boxes the underlying error into [`BoxedError`].
51///
52/// This is mainly used by flownode to invoke a frontend instance in standalone mode.
53#[async_trait::async_trait]
54pub trait GrpcQueryHandlerWithBoxedError: Send + Sync + 'static {
55    async fn do_query(
56        &self,
57        query: Request,
58        ctx: QueryContextRef,
59    ) -> std::result::Result<Output, BoxedError>;
60}
61
62/// auto impl
63#[async_trait::async_trait]
64impl<T: GrpcQueryHandler + Send + Sync + 'static> GrpcQueryHandlerWithBoxedError for T {
65    async fn do_query(
66        &self,
67        query: Request,
68        ctx: QueryContextRef,
69    ) -> std::result::Result<Output, BoxedError> {
70        self.do_query(query, ctx).await.map_err(BoxedError::new)
71    }
72}

74#[derive(Debug, Clone)]
75pub struct HandlerMutable {
76    handler: Arc<Mutex<Option<Weak<dyn GrpcQueryHandlerWithBoxedError>>>>,
77    is_initialized: Arc<SetOnce<()>>,
78}
79
80impl HandlerMutable {
81    pub async fn set_handler(&self, handler: Weak<dyn GrpcQueryHandlerWithBoxedError>) {
82        *self.handler.lock().unwrap() = Some(handler);
83        // Ignore the error, as we allow the handler to be set multiple times.
84        let _ = self.is_initialized.set(());
85    }
86}

88/// A simple frontend client able to execute sql using grpc protocol
89///
90/// This is for computation-heavy query which need to offload computation to frontend, lifting the load from flownode
91#[derive(Debug, Clone)]
92pub enum FrontendClient {
93    Distributed {
94        meta_client: Arc<MetaClient>,
95        chnl_mgr: ChannelManager,
96        query: QueryOptions,
97        batch_opts: BatchingModeOptions,
98    },
99    Standalone {
100        /// for the sake of simplicity still use grpc even in standalone mode
101        /// notice the client here should all be lazy, so that can wait after frontend is booted then make conn
102        database_client: HandlerMutable,
103        query: QueryOptions,
104    },
105}
106
107impl FrontendClient {
108    /// Create a new empty frontend client, with a `HandlerMutable` to set the grpc handler later
109    pub fn from_empty_grpc_handler(query: QueryOptions) -> (Self, HandlerMutable) {
110        let is_initialized = Arc::new(SetOnce::new());
111        let handler = HandlerMutable {
112            handler: Arc::new(Mutex::new(None)),
113            is_initialized,
114        };
115        (
116            Self::Standalone {
117                database_client: handler.clone(),
118                query,
119            },
120            handler,
121        )
122    }
123
124    /// Waits until the frontend client is initialized.
125    pub async fn wait_initialized(&self) {
126        if let FrontendClient::Standalone {
127            database_client, ..
128        } = self
129        {
130            database_client.is_initialized.wait().await;
131        }
132    }
133
134    pub fn from_meta_client(
135        meta_client: Arc<MetaClient>,
136        query: QueryOptions,
137        batch_opts: BatchingModeOptions,
138    ) -> Result<Self, Error> {
139        common_telemetry::info!("Frontend client build without auth");
140        Ok(Self::Distributed {
141            meta_client,
142            chnl_mgr: {
143                let cfg = ChannelConfig::new()
144                    .connect_timeout(batch_opts.grpc_conn_timeout)
145                    .timeout(Some(batch_opts.query_timeout));
146
147                let tls_config = load_client_tls_config(batch_opts.frontend_tls.clone())
148                    .context(InvalidClientConfigSnafu)?;
149                ChannelManager::with_config(cfg, tls_config)
150            },
151            query,
152            batch_opts,
153        })
154    }
155
156    pub fn from_grpc_handler(
157        grpc_handler: Weak<dyn GrpcQueryHandlerWithBoxedError>,
158        query: QueryOptions,
159    ) -> Self {
160        let is_initialized = Arc::new(SetOnce::new_with(Some(())));
161        let handler = HandlerMutable {
162            handler: Arc::new(Mutex::new(Some(grpc_handler))),
163            is_initialized: is_initialized.clone(),
164        };
165
166        Self::Standalone {
167            database_client: handler,
168            query,
169        }
170    }
171}

184    /// Try sending a "SELECT 1" to the database
185    async fn try_select_one(&self) -> Result<(), Error> {
186        // notice here use `sql` for `SELECT 1` return 1 row
187        let _ = self
188            .database
189            .sql("SELECT 1")
190            .await
191            .with_context(|_| InvalidRequestSnafu {
192                context: format!("Failed to handle `SELECT 1` request at {:?}", self.peer),
193            })?;
194        Ok(())
195    }

198impl FrontendClient {
199    /// scan for available frontend from metadata
200    pub(crate) async fn scan_for_frontend(&self) -> Result<Vec<Peer>, Error> {
201        let Self::Distributed { meta_client, .. } = self else {
202            return Ok(vec![]);
203        };
204
205        meta_client
206            .active_frontends()
207            .await
208            .map(|nodes| nodes.into_iter().map(|node| node.peer).collect())
209            .map_err(BoxedError::new)
210            .context(ExternalSnafu)
211    }

213    /// Probes all discovered frontends without auth.
214    ///
215    /// Returns non-auth failures to allow callers to retry transient connectivity
216    /// errors. Authentication failures are returned immediately because they mean
217    /// a frontend advertised an auth-protected endpoint to flownodes.
218    pub(crate) async fn check_all_frontends_without_auth(
219        &self,
220        frontends: &[Peer],
221    ) -> Result<Vec<String>, Error> {
222        let Self::Distributed {
223            chnl_mgr,
224            batch_opts,
225            ..
226        } = self
227        else {
228            return Ok(vec![]);
229        };
230
231        let probe_timeout = batch_opts.grpc_conn_timeout;
232        let mut probes = frontends
233            .iter()
234            .map(|peer| {
235                let addr = peer.addr.clone();
236                let chnl_mgr = chnl_mgr.clone();
237
238                async move {
239                    let client = Client::with_manager_and_urls(chnl_mgr, vec![addr.clone()]);
240                    let database = Database::new(DEFAULT_CATALOG_NAME, DEFAULT_SCHEMA_NAME, client);
241
242                    match tokio::time::timeout(probe_timeout, database.sql("SELECT 1")).await {
243                        Ok(Ok(_)) => Ok(None),
244                        Ok(Err(err)) if err.tonic_code() == Some(tonic::Code::Unauthenticated) => {
245                            Err(err).context(InvalidRequestSnafu {
246                                context: format!(
247                                    "Frontend {addr} rejected unauthenticated flownode probe; ensure frontend internal_grpc is advertised to metasrv"
248                                ),
249                            })
250                        }
251                        Ok(Err(err)) => Ok(Some(format!("{addr}: {err}"))),
252                        Err(_) => Ok(Some(format!(
253                            "{addr}: health check timed out after {probe_timeout:?}"
254                        ))),
255                    }
256                }
257            })
258            .collect::<FuturesUnordered<_>>();
259
260        let mut failures = Vec::new();
261        while let Some(probe_result) = probes.next().await {
262            if let Some(failure) = probe_result? {
263                failures.push(failure);
264            }
265        }
266
267        Ok(failures)
268    }

270    /// Get a frontend discovered by metasrv and verified with a query probe.
271    async fn get_random_active_frontend(
272        &self,
273        catalog: &str,
274        schema: &str,
275    ) -> Result<DatabaseWithPeer, Error> {
276        let Self::Distributed {
277            meta_client: _,
278            chnl_mgr,
279            query: _,
280            batch_opts,
281        } = self
282        else {
283            return UnexpectedSnafu {
284                reason: "Expect distributed mode",
285            }
286            .fail();
287        };
288
289        let mut interval = tokio::time::interval(batch_opts.grpc_conn_timeout);
290        interval.tick().await;
291        for retry in 0..batch_opts.experimental_grpc_max_retries {
292            let mut frontends = self.scan_for_frontend().await?;
293            // shuffle the frontends to avoid always pick the same one
294            frontends.shuffle(&mut rng());
295
296            for peer in frontends {
297                let addr = peer.addr.clone();
298                let client = Client::with_manager_and_urls(chnl_mgr.clone(), vec![addr.clone()]);
299                let database = Database::new(catalog, schema, client);
300                let db = DatabaseWithPeer::new(database, peer);
301                match db.try_select_one().await {
302                    Ok(_) => return Ok(db),
303                    Err(e) => {
304                        warn!(
305                            "Failed to connect to frontend {} on retry={}: \n{e:?}",
306                            addr, retry
307                        );
308                    }
309                }
310            }
311            // no available frontend
312            // sleep and retry
313            interval.tick().await;
314        }
315
316        NoAvailableFrontendSnafu {
317            timeout: batch_opts.grpc_conn_timeout,
318            context: "No available frontend found that is able to process query",
319        }
320        .fail()
321    }

344    /// Execute a SQL statement on the frontend.
345    pub async fn sql(&self, catalog: &str, schema: &str, sql: &str) -> Result<Output, Error> {
346        match self {
347            FrontendClient::Distributed { .. } => {
348                let db = self.get_random_active_frontend(catalog, schema).await?;
349                db.database
350                    .sql(sql)
351                    .await
352                    .map_err(BoxedError::new)
353                    .context(ExternalSnafu)
354            }
355            FrontendClient::Standalone {
356                database_client, ..
357            } => {
358                let ctx = QueryContextBuilder::default()
359                    .current_catalog(catalog.to_string())
360                    .current_schema(schema.to_string())
361                    .build();
362                let ctx = Arc::new(ctx);
363                {
364                    let database_client = {
365                        database_client
366                            .handler
367                            .lock()
368                            .unwrap()
369                            .as_ref()
370                            .context(UnexpectedSnafu {
371                                reason: "Standalone's frontend instance is not set",
372                            })?
373                            .upgrade()
374                            .context(UnexpectedSnafu {
375                                reason: "Failed to upgrade database client",
376                            })?
377                    };
378                    let req = Request::Query(QueryRequest {
379                        query: Some(Query::Sql(sql.to_string())),
380                    });
381                    database_client
382                        .do_query(req, ctx)
383                        .await
384                        .map_err(BoxedError::new)
385                        .context(ExternalSnafu)
386                }
387            }
388        }
389    }

391    pub(crate) async fn query_with_terminal_metrics(
392        &self,
393        catalog: &str,
394        schema: &str,
395        request: QueryRequest,
396        extensions: &[(&str, &str)],
397        peer_desc: &mut Option<PeerDesc>,
398    ) -> Result<OutputWithMetrics, Error> {
399        let flow_extensions = build_flow_extensions(extensions)?;
400        match self {
401            FrontendClient::Distributed {
402                query, batch_opts, ..
403            } => {
404                let query_parallelism = query.parallelism.to_string();
405                let hints = vec![
406                    (QUERY_PARALLELISM_HINT, query_parallelism.as_str()),
407                    (READ_PREFERENCE_HINT, batch_opts.read_preference.as_ref()),
408                ];
409                let db = self.get_random_active_frontend(catalog, schema).await?;
410                *peer_desc = Some(PeerDesc::Dist {
411                    peer: db.peer.clone(),
412                });
413                db.database
414                    .query_with_terminal_metrics_and_flow_extensions(
415                        request,
416                        &hints,
417                        extensions,
418                        &Default::default(),
419                    )
420                    .await
421                    .map_err(BoxedError::new)
422                    .context(ExternalSnafu)
423            }
424            FrontendClient::Standalone {
425                database_client,
426                query,
427            } => {
428                *peer_desc = Some(PeerDesc::Standalone);
429                let mut extensions_map = HashMap::from([(
430                    QUERY_PARALLELISM_HINT.to_string(),
431                    query.parallelism.to_string(),
432                )]);
433                for (key, value) in extensions {
434                    extensions_map.insert((*key).to_string(), (*value).to_string());
435                }
436                let ctx = QueryContextBuilder::default()
437                    .current_catalog(catalog.to_string())
438                    .current_schema(schema.to_string())
439                    .extensions(extensions_map)
440                    .build();
441                let ctx = Arc::new(ctx);
442                let database_client = {
443                    database_client
444                        .handler
445                        .lock()
446                        .map_err(|e| {
447                            UnexpectedSnafu {
448                                reason: format!("Failed to lock database client: {e}"),
449                            }
450                            .build()
451                        })?
452                        .as_ref()
453                        .context(UnexpectedSnafu {
454                            reason: "Standalone's frontend instance is not set",
455                        })?
456                        .upgrade()
457                        .context(UnexpectedSnafu {
458                            reason: "Failed to upgrade database client",
459                        })?
460                };
461                database_client
462                    .do_query(Request::Query(request), ctx.clone())
463                    .await
464                    .map(|output| {
465                        wrap_standalone_output_with_terminal_metrics(output, &flow_extensions, &ctx)
466                    })
467                    .map_err(BoxedError::new)
468                    .context(ExternalSnafu)
469            }
470        }
471    }

473    /// Handle a request to frontend
474    pub(crate) async fn handle(
475        &self,
476        req: api::v1::greptime_request::Request,
477        catalog: &str,
478        schema: &str,
479        peer_desc: &mut Option<PeerDesc>,
480    ) -> Result<u32, Error> {
481        match self {
482            FrontendClient::Distributed {
483                query, batch_opts, ..
484            } => {
485                let db = self.get_random_active_frontend(catalog, schema).await?;
486
487                *peer_desc = Some(PeerDesc::Dist {
488                    peer: db.peer.clone(),
489                });
490
491                db.database
492                    .handle_with_retry(
493                        req.clone(),
494                        batch_opts.experimental_grpc_max_retries,
495                        &[
496                            (QUERY_PARALLELISM_HINT, &query.parallelism.to_string()),
497                            (READ_PREFERENCE_HINT, batch_opts.read_preference.as_ref()),
498                        ],
499                    )
500                    .await
501                    .with_context(|_| InvalidRequestSnafu {
502                        context: format!("Failed to handle request at {:?}: {:?}", db.peer, req),
503                    })
504            }
505            FrontendClient::Standalone {
506                database_client,
507                query,
508            } => {
509                let ctx = QueryContextBuilder::default()
510                    .current_catalog(catalog.to_string())
511                    .current_schema(schema.to_string())
512                    .extensions(HashMap::from([(
513                        QUERY_PARALLELISM_HINT.to_string(),
514                        query.parallelism.to_string(),
515                    )]))
516                    .build();
517                let ctx = Arc::new(ctx);
518                {
519                    let database_client = {
520                        database_client
521                            .handler
522                            .lock()
523                            .unwrap()
524                            .as_ref()
525                            .context(UnexpectedSnafu {
526                                reason: "Standalone's frontend instance is not set",
527                            })?
528                            .upgrade()
529                            .context(UnexpectedSnafu {
530                                reason: "Failed to upgrade database client",
531                            })?
532                    };
533                    let resp: common_query::Output = database_client
534                        .do_query(req, ctx)
535                        .await
536                        .map_err(BoxedError::new)
537                        .context(ExternalSnafu)?;
538                    match resp.data {
539                        common_query::OutputData::AffectedRows(rows) => {
540                            Ok(rows.try_into().map_err(|_| {
541                                UnexpectedSnafu {
542                                    reason: format!("Failed to convert rows to u32: {}", rows),
543                                }
544                                .build()
545                            })?)
546                        }
547                        _ => UnexpectedSnafu {
548                            reason: "Unexpected output data",
549                        }
550                        .fail(),
551                    }
552                }
553            }
554        }
555    }
556}

558fn build_flow_extensions(extensions: &[(&str, &str)]) -> Result<FlowQueryExtensions, Error> {
559    let flow_extensions = HashMap::from_iter(
560        extensions
561            .iter()
562            .map(|(key, value)| ((*key).to_string(), (*value).to_string())),
563    );
564    FlowQueryExtensions::parse_flow_extensions(&flow_extensions)
565        .map_err(BoxedError::new)
566        .context(ExternalSnafu)
567        .map(|extensions| extensions.unwrap_or_default())
568}

570fn wrap_standalone_output_with_terminal_metrics(
571    output: Output,
572    flow_extensions: &FlowQueryExtensions,
573    query_ctx: &QueryContextRef,
574) -> OutputWithMetrics {
575    let should_collect_region_watermark = flow_extensions.should_collect_region_watermark();
576    let terminal_metrics =
577        if should_collect_region_watermark && !matches!(&output.data, OutputData::Stream(_)) {
578            output
579                .meta
580                .plan
581                .clone()
582                .and_then(terminal_recordbatch_metrics_from_plan)
583                .or_else(|| terminal_recordbatch_metrics_from_snapshots(query_ctx))
584        } else {
585            None
586        };
587    let result = OutputWithMetrics::from_output(output);
588    if let Some(metrics) = terminal_metrics {
589        result.metrics.update(Some(metrics));
590    }
591    result
592}
593
594fn terminal_recordbatch_metrics_from_snapshots(
595    query_ctx: &QueryContextRef,
596) -> Option<RecordBatchMetrics> {
597    let mut region_watermarks = query_ctx
598        .snapshots()
599        .into_iter()
600        .map(|(region_id, watermark)| RegionWatermarkEntry {
601            region_id,
602            watermark: Some(watermark),
603        })
604        .collect::<Vec<_>>();
605    if region_watermarks.is_empty() {
606        return None;
607    }
608
609    region_watermarks.sort_by_key(|entry| entry.region_id);
610    Some(RecordBatchMetrics {
611        region_watermarks,
612        ..Default::default()
613    })
614}

616/// Describe a peer of frontend
617#[derive(Debug, Default, Clone)]
618pub(crate) enum PeerDesc {
619    /// The query failed before a frontend peer was selected.
620    #[default]
621    Unknown,
622    /// Distributed mode's frontend peer address
623    Dist {
624        /// frontend peer address
625        peer: Peer,
626    },
627    /// Standalone mode
628    Standalone,
629}
630
631impl std::fmt::Display for PeerDesc {
632    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
633        match self {
634            PeerDesc::Unknown => write!(f, "unknown"),
635            PeerDesc::Dist { peer } => write!(f, "{}", peer.addr),
636            PeerDesc::Standalone => write!(f, "standalone"),
637        }
638    }
639}

831        let listener = TcpListener::bind("127.0.0.1:0")
832            .await
833            .expect("bind test flight server");
834        let addr = listener.local_addr().expect("local addr").to_string();
835        let server = tokio::spawn(async move {
836            tonic::transport::Server::builder()
837                .add_service(FlightServiceServer::new(FlightCraftWrapper(handler)))
838                .serve_with_incoming(TcpListenerStream::new(listener))
839                .await
840                .expect("serve test flight server");
841        });
842
843        (addr, server)
844    }

1062    async fn test_check_all_frontends_without_auth_checks_frontends_concurrently() {
1063        let barrier = Arc::new(tokio::sync::Barrier::new(2));
1064        let (addr1, server1) = start_flight_server(WaitForConcurrentFlight {
1065            barrier: barrier.clone(),
1066        })
1067        .await;
1068        let (addr2, server2) = start_flight_server(WaitForConcurrentFlight { barrier }).await;
1069        let client = FrontendClient::from_meta_client(
1070            Arc::new(MetaClient::new(0, api::v1::meta::Role::Frontend)),
1071            QueryOptions::default(),
1072            BatchingModeOptions {
1073                grpc_conn_timeout: Duration::from_millis(500),
1074                ..Default::default()
1075            },
1076        )
1077        .unwrap();
1078
1079        let failures = timeout(
1080            Duration::from_secs(2),
1081            client.check_all_frontends_without_auth(&[
1082                Peer {
1083                    id: 1,
1084                    addr: addr1.clone(),
1085                },
1086                Peer {
1087                    id: 2,
1088                    addr: addr2.clone(),
1089                },
1090            ]),
1091        )
1092        .await
1093        .expect("concurrent probes should complete before per-peer timeouts")
1094        .unwrap();
1095        server1.abort();
1096        server2.abort();
1097
1098        assert_eq!(failures.len(), 2);
1099        assert!(failures.iter().any(|failure| failure.contains(&addr1)));
1100        assert!(failures.iter().any(|failure| failure.contains(&addr2)));
1101        assert!(
1102            failures
1103                .iter()
1104                .all(|failure| !failure.contains("health check timed out")),
1105            "sequential probes would time out before both requests reach the barrier: {failures:?}"
1106        );
